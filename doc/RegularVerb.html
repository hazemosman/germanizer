def present_conjugate
        if defined?(@stem)
                ["ich #{@stem}e",
                "du #{@stem}st",
                "er,sie, es #{@stem}t",
                "wir #{@stem}en",
                "ihr #{@stem}t",
                "sie,Sie #{@stem}t"]
        else
                puts "Could not work with the verb: #{@infinitive}"
        end
end
