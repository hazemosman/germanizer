def present_conjugate
        if defined?(@stem)
                # TODO Get irregular verbs from DB
                case @infinitive
                 when 'sein'
                     present_conjugate_sein
                 when 'haben'
                     present_conjugate_haben
                 else
                     super
                 end
        else
                puts "Could not work with the verb: #{@infinitive}"
        end
end
